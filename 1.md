
## 03 .PROBLEM STATEMENT:
Consider a friend’s network on Facebook social web site. Model it as a graph to represent each
node as a user and a link to represent the friend relationship between them using adjacency list
representation and perform DFS & BFS traversals .
a. Create the Graph using Adjacency list ( Friend name , ID)
b. Display the Graph
c. Perform the recursive and non-recursive Traversal using Depth First Traversal (Using ID)
d. Perform the non-recursive Traversal using Breadth First Traversal (Using ID
```cpp
#include <iostream>
#include <stack>
#include <queue>
using namespace std;

class Graph {
    struct Node {
        int vertex;
        Node* next;
    };

    Node* head[20];
    int n;
    string name[20];

public:

    // Constructor
    Graph() {
        cout << "Enter number of users: ";
        cin >> n;

        for (int i = 0; i < n; i++) {
            head[i] = NULL;
        }

        for (int i = 0; i < n; i++) {
            cout << "Enter name of user " << i << ": ";
            cin >> name[i];
        }
    }

    // Create graph (Adjacency List)
    void create() {
        char ch;

        for (int i = 0; i < n; i++) {
            Node* temp = NULL;

            do {
                int v;
                cout << "Enter friend of " << i << " (ID): ";
                cin >> v;

                if (v == i) {
                    cout << "Self loop not allowed\n";
                } else {
                    Node* newNode = new Node;
                    newNode->vertex = v;
                    newNode->next = NULL;

                    if (head[i] == NULL) {
                        head[i] = newNode;
                    } else {
                        temp = head[i];
                        while (temp->next != NULL)
                            temp = temp->next;
                        temp->next = newNode;
                    }
                }

                cout << "Add more friends for " << i << "? (y/n): ";
                cin >> ch;

            } while (ch == 'y' || ch == 'Y');
        }
    }

    // Display graph
    void display() {
        cout << "\nAdjacency List:\n";
        for (int i = 0; i < n; i++) {
            cout << i << " (" << name[i] << ") -> ";
            Node* temp = head[i];

            while (temp != NULL) {
                cout << temp->vertex << " ";
                temp = temp->next;
            }
            cout << endl;
        }
    }

    // Recursive DFS
    void dfsRec(int v, int visited[]) {
        cout << v << " ";
        visited[v] = 1;

        Node* temp = head[v];
        while (temp != NULL) {
            if (!visited[temp->vertex]) {
                dfsRec(temp->vertex, visited);
            }
            temp = temp->next;
        }
    }

    void startDFSRec(int start) {
        int visited[20] = {0};
        cout << "\nDFS Recursive: ";
        dfsRec(start, visited);
    }

    // Non-recursive DFS (as per PPT stack algo)
    void dfsNonRec(int start) {
        int visited[20] = {0};
        stack<int> st;

        st.push(start);
        visited[start] = 1;

        cout << "\nDFS Non-Recursive: ";

        while (!st.empty()) {
            int v = st.top();
            st.pop();
            cout << v << " ";

            Node* temp = head[v];
            while (temp != NULL) {
                if (!visited[temp->vertex]) {
                    st.push(temp->vertex);
                    visited[temp->vertex] = 1;
                }
                temp = temp->next;
            }
        }
    }

    // BFS (as per PPT queue algo)
    void bfs(int start) {
        int visited[20] = {0};
        queue<int> q;

        q.push(start);
        visited[start] = 1;

        cout << "\nBFS: ";

        while (!q.empty()) {
            int v = q.front();
            q.pop();
            cout << v << " ";

            Node* temp = head[v];
            while (temp != NULL) {
                if (!visited[temp->vertex]) {
                    q.push(temp->vertex);
                    visited[temp->vertex] = 1;
                }
                temp = temp->next;
            }
        }
    }
};

int main() {
    Graph g;

    g.create();
    g.display();

    int start;
    cout << "\nEnter starting ID: ";
    cin >> start;

    g.startDFSRec(start);
    g.dfsNonRec(start);
    g.bfs(start);

    return 0;
    }
```

```cpp
int prims(int cost[10][10], int n, int start_v)
{
   int nearest[10];
   int t[10][2];      // Stores MST edges
   int mincost = 0;
   int i, j, k;


   // Initialize nearest[]
   for(i = 0; i < n; i++)
       nearest[i] = start_v;


   nearest[start_v] = -1;   // Include start vertex in MST


   // Find n-1 edges
   for(i = 0; i < n-1; i++)
   {
       int min = 9999;
       int v = -1;


       // Find minimum edge
       for(k = 0; k < n; k++)
       {
           if(nearest[k] != -1 && cost[k][nearest[k]] < min)
           {
               min = cost[k][nearest[k]];
               v = k;
           }
       }


       // Store edge
       t[i][0] = v;
       t[i][1] = nearest[v];
       mincost += cost[v][nearest[v]];


       nearest[v] = -1;   // Include vertex in MST


       // Update nearest[]
       for(k = 0; k < n; k++)
       {
           if(nearest[k] != -1 && cost[k][v] < cost[k][nearest[k]])
               nearest[k] = v;
       }
   }

   cout << "\nEdges in MST:\n";
   for(i = 0; i < n-1; i++)
       cout << t[i][0] << " - " << t[i][1] << endl;


   return mincost;
}
```

## 05. PROBLEM STATEMENT:
Write a program to implement AVL tree for Dictionary. A Dictionary stores keywords &
its meaning. Provide facility for adding new keywords, updating meaning of any entry. Provide a
facility to display whole data sorted in ascending order. Also find how many maximum
comparisons may require for finding any keyword. Use a Height balanced tree and find the
complexity for finding a keyword.

```cpp
#include<iostream>
#include<string>
#include<algorithm>   // for max()
using namespace std;

class node {
public:
    string data;
    node* left;
    node* right;

    node(string key) {
        data = key;
        left = right = NULL;
    }
};

class AVL {
public:
    node* root;

    AVL() {
        root = NULL;
    }

    void create_avl();
    node* insert(node* root, string key);
    node* LL(node* t);
    node* RR(node* t);
    node* LR(node* t);
    node* RL(node* t);
    node* Balance(node* t);
    int BF(node* t);
    int ht(node* t);
    void inorder(node* root);
};

// Create AVL
void AVL::create_avl() {
    string key;
    char ch;

    do {
        cout << "Enter Key: ";
        cin >> key;

        root = insert(root, key);

        cout << "Do you want to continue? (y/n): ";
        cin >> ch;

    } while (ch == 'y' || ch == 'Y');
}

// Insert node
node* AVL::insert(node* root, string key) {
    if (root == NULL) {
        return new node(key);
    }

    if (key < root->data) {   // ASCII comparison
        root->left = insert(root->left, key);
    }
    else {
        root->right = insert(root->right, key);
    }

    return Balance(root);
}

node* AVL::Balance(node* t) {
    int bf = BF(t);

    // Left heavy
    if (bf > 1) {
        if (BF(t->left) >= 0)
            return LL(t);
        else
            return LR(t);
    }

    // Right heavy
    if (bf < -1) {
        if (BF(t->right) <= 0)
            return RR(t);
        else
            return RL(t);
    }

    return t;
}

// Rotations
node* AVL::LL(node* t) {
    node* y = t->left;
    t->left = y->right;
    y->right = t;
    return y;
}

node* AVL::RR(node* t) {
    node* y = t->right;
    t->right = y->left;
    y->left = t;
    return y;
}

node* AVL::LR(node* t) {
    t->left = RR(t->left);
    return LL(t);
}

node* AVL::RL(node* t) {
    t->right = LL(t->right);
    return RR(t);
}

// Balance Factor
int AVL::BF(node* t) {
    if (t == NULL)
        return 0;
    return ht(t->left) - ht(t->right);
}

// Height
int AVL::ht(node* t) {
    if (t == NULL)
        return 0;
    return max(ht(t->left), ht(t->right)) + 1;
}

// Inorder traversal
void AVL::inorder(node* root) {
    if (root != NULL) {
        inorder(root->left);
        cout << root->data << " ";
        inorder(root->right);
    }
}

// Main
int main() {
    AVL tree;

    tree.create_avl();

    cout << "\nInorder Traversal: ";
    tree.inorder(tree.root);

    return 0;
}
```

## 06. PROBLEM STATMENT: N QUEENS 
```cpp
#include <iostream>
using namespace std;

#define N 4   // you can change size

// Function to print board
void printBoard(int board[N][N]) {
    for(int i = 0; i < N; i++) {
        for(int j = 0; j < N; j++) {
            cout << board[i][j] << " ";
        }
        cout << endl;
    }
}

// Check if safe to place queen
bool isSafe(int board[N][N], int row, int col) {

    // Check column
    for(int i = 0; i < row; i++) {
        if(board[i][col] == 1)
            return false;
    }

    // Left diagonal
    for(int i = row, j = col; i >= 0 && j >= 0; i--, j--) {
        if(board[i][j] == 1)
            return false;
    }

    // Right diagonal
    for(int i = row, j = col; i >= 0 && j < N; i--, j++) {
        if(board[i][j] == 1)
            return false;
    }

    return true;
}

// Backtracking function
bool solve(int board[N][N], int row) {

    if(row == N) {
        printBoard(board);
        return true;
    }

    for(int col = 0; col < N; col++) {
        if(isSafe(board, row, col)) {

            board[row][col] = 1;

            if(solve(board, row + 1))
                return true;

            // Backtrack
            board[row][col] = 0;
        }
    }

    return false;
}

int main() {
    int board[N][N] = {0};

    if(!solve(board, 0))
        cout << "No solution exists";

    return 0;
}
```

## 07. PROBLEM STATEMENT:
Implement 0/1 Knapsack OR Traveling Salesperson problem using Dynamic Programming

```cpp
#include <iostream>
#include <iomanip>    //Input-Output Manipulation [only for spacing not imp]
using namespace std;

int main()
{
    int n, W;

    cout << "Enter number of items: ";
    cin >> n;

    int weight[n + 1], value[n + 1];

    cout << "Enter weights of items:\n";
    for (int i = 1; i <= n; i++)
        cin >> weight[i];

    cout << "Enter values of items:\n";
    for (int i = 1; i <= n; i++)
        cin >> value[i];

    cout << "Enter knapsack capacity: ";
    cin >> W;

    int B[n + 1][W + 1];

    for (int w = 0; w <= W; w++)
        B[0][w] = 0;

    for (int i = 1; i <= n; i++)
        B[i][0] = 0;


    for (int i = 1; i <= n; i++)
    {
        for (int w = 0; w <= W; w++)
        {
            if (weight[i] <= w)
            {
                B[i][w] = max(value[i] + B[i - 1][w - weight[i]], B[i - 1][w]);
            }
            else
            {
                B[i][w] = B[i - 1][w];
            }
        }
    }

    cout << "\nDP Table:\n\n";


    cout << "i/w ";
    for (int w = 0; w <= W; w++)
        cout << setw(5) << w;
    cout << endl;

    for (int i = 0; i <= n; i++)
    {
        cout << setw(3) << i << " ";
        for (int w = 0; w <= W; w++)
        {
            cout << setw(5) << B[i][w];
        }
        cout << endl;
    }

    cout << "\nMaximum value = " << B[n][W] << endl;

    return 0;
}
```

## 08. PROBLEM STATEMENT 
Implement Direct access file using hashing (linear probing with and without replacement)
perform following operations on it a) Create Student Database[ Roll no, Name, Marks] b)
Display Database c) Add a record d) Search a record e) Modify a record

```CPP
#include <iostream>
using namespace std;

#define SIZE 10

struct Student {
    int roll;
    string name;
    int marks;
};

Student table[SIZE];

// Initialize table
void init() {
    for(int i = 0; i < SIZE; i++)
        table[i].roll = -1;
}

// Hash function
int hashFunc(int key) {
    return key % SIZE;
}

// Insert record
void insertRecord() {
    Student s;
    cout << "Enter Roll, Name, Marks: ";
    cin >> s.roll >> s.name >> s.marks;

    int index = hashFunc(s.roll);

    while(table[index].roll != -1) {
        index = (index + 1) % SIZE;
    }

    table[index] = s;
    cout << "Record Inserted\n";
}

// Display table
void display() {
    cout << "\nHash Table:\n";
    for(int i = 0; i < SIZE; i++) {
        if(table[i].roll != -1)
            cout << i << " -> " << table[i].roll << " "
                 << table[i].name << " "
                 << table[i].marks << endl;
        else
            cout << i << " -> EMPTY\n";
    }
}

// Search record
int searchRecord(int key) {
    int index = hashFunc(key);
    int start = index;

    while(table[index].roll != -1) {
        if(table[index].roll == key)
            return index;

        index = (index + 1) % SIZE;

        if(index == start)
            break;
    }
    return -1;
}

// Modify record
void modifyRecord() {
    int key;
    cout << "Enter roll to modify: ";
    cin >> key;

    int pos = searchRecord(key);

    if(pos == -1) {
        cout << "Record not found\n";
        return;
    }

    cout << "Enter new Name and Marks: ";
    cin >> table[pos].name >> table[pos].marks;

    cout << "Record Updated\n";
}

int main() {
    init();
    int choice, key, pos;

    do {
        cout << "\n--- MENU ---\n";
        cout << "1. Insert\n2. Display\n3. Search\n4. Modify\n5. Exit\n";
        cout << "Enter choice: ";
        cin >> choice;

        switch(choice) {

            case 1:
                insertRecord();
                break;

            case 2:
                display();
                break;

            case 3:
                cout << "Enter roll to search: ";
                cin >> key;
                pos = searchRecord(key);

                if(pos == -1)
                    cout << "Record not found\n";
                else
                    cout << "Record found at index " << pos << endl;
                break;

            case 4:
                modifyRecord();
                break;

            case 5:
                cout << "Exiting...\n";
                break;

            default:
                cout << "Invalid choice\n";
        }

    } while(choice != 5);

    return 0;
}
```
